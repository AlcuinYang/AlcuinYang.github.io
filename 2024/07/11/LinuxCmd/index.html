LinuxCmd

以下是为您整理的以 .md 格式呈现的 Linux 常用命令文档：

    free [选项]
    
    free -h
    
    ps [选项]
    
    ps -ef
    
    top
    
    ifconfig [选项]
    
    ifconfig -a
    
    tcpdump [选项] [表达式]
    
    tcpdump -i eth0 -c 100
    
    netstat [选项]
    
    netstat -tun
    
    ping [选项] 目标主机
    
    ping -c 5 www.baidu.com
    
    traceroute 目标主机
    
    traceroute www.baidu.com
    
    route [操作] [参数]
    
    route add default gw 192.168.1.1
    
    pwd
    
    cd [目录路径]
    
    cd /home/user
    
    ls [选项] [目录路径]
    
    ls -lah
    
    mkdir [选项] 目录路径
    
    mkdir -p dir1/dir2
    
    cp [选项] 源文件或目录 目标文件或目录
    
    cp -r source_dir destination_dir
    
    rm [选项] 文件或目录
    
    rm -r dir1
    
    mv [选项] 源文件或目录 目标文件或目录
    
    mv file1 file2
    
    ln [选项] 源文件 目标文件
    
    ln -s file1 link1
    
    cat [文件]
    
    cat file.txt
    
    more [文件]
    
    more file.txt
    
    less [文件]
    
    less file.txt
    
    head [选项] [文件]
    
    head -n 20 file.txt
    
    tail [选项] [文件]
    
    tail -n 15 file.txt
    
    grep [选项] 模式 [文件]
    
    grep -i "hello" file.txt
    
    tar [选项] [归档文件名] [文件或目录]
    
    useradd [选项] 用户名
    
    useradd -u 1001 -g users -d /home/newuser newuser
    
    usermod [选项] 用户名
    
    usermod -u 1002 olduser
    
    userdel [选项] 用户名
    
    userdel -r user1
    
    groupadd [选项] 组名
    
    groupadd mygroup
    
    groupmod [选项] 组名
    
    groupmod -g 1005 mygroup
    
    groupdel 组名
    
    groupdel mygroup
    
    chmod [选项] 权限模式 文件或目录
    
    chmod 755 file.txt
    
    chown [选项] 所有者[:组] 文件或目录
    
    chown user1:group1 file.txt
